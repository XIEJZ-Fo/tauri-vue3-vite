1. 系统总体设计视图

1.1 物理架构

•   部署模型：单节点、本地化桌面应用程序，由Tauri框架打包为一个可执行文件，实现NF-005.1要求。

•   运行时：应用程序启动后，Vue 3 单页应用在本地浏览器内核中运行，所有数据处理、存储、UI交互均在用户设备上完成，满足NF-002.1离线操作要求。

1.2 逻辑架构

采用扁平化模块架构，核心模块通过app.vue进行集成与协调，符合“外观模式”（Facade Pattern）思想，以降低耦合度。
•   todolist.vue：核心数据模型与业务逻辑中心，对外提供精简的taskService接口。

•   pomodoro.vue：计时引擎，通过依赖注入获取taskService，实现状态联动。

•   stats.vue：数据计算与可视化模块，从storage.ts拉取数据。

•   storage.ts：数据持久化与一致性保障层，为所有模块提供统一、安全的读写接口。

•   app.vue：应用入口、布局框架与模块协调器。

在模块的 <script setup> 中定义核心数据结构和响应式状态。
// Task 状态枚举
const TaskStatus = {
  NOT_STARTED: 'not_started',
  IN_PROGRESS: 'in_progress', // 与番茄钟关联时
  COMPLETED: 'completed'
};

// 父任务类 (Category)
class ParentTask {
  constructor({ id, name, description, priority, dueDate }) {
    this.id = id || generateId();
    this.name = name;
    this.description = description;
    this.priority = priority; // 例如：'high', 'medium', 'low'
    this.dueDate = dueDate; // Date 对象
    this.createdAt = new Date();
    this.children = []; // 包含的子任务对象数组
    this.status = TaskStatus.NOT_STARTED; // 由子任务状态衍生
  }
  // 计算属性：总番茄数（所有子任务之和）
  get totalPomodoros() {
    return this.children.reduce((sum, child) => sum + child.completedPomodoros, 0);
  }
  // 计算属性：状态（若任一子任务为“进行中”，则父任务为“进行中”）
  get derivedStatus() {
    if (this.children.some(c => c.status === TaskStatus.IN_PROGRESS)) return TaskStatus.IN_PROGRESS;
    if (this.children.every(c => c.status === TaskStatus.COMPLETED)) return TaskStatus.COMPLETED;
    return TaskStatus.NOT_STARTED;
  }
}

// 子任务类
class ChildTask {
  constructor({ id, parentId, name, description, priority, dueDate }) {
    this.id = id || generateId();
    this.parentId = parentId; // 所属父任务的ID
    this.name = name;
    this.description = description;
    this.priority = priority;
    this.dueDate = dueDate;
    this.createdAt = new Date();
    this.completedPomodoros = 0; // 实际完成的番茄钟数量
    this.status = TaskStatus.NOT_STARTED;
  }
}

// 响应式状态
import { ref, reactive, computed } from 'vue';
const parentTasks = ref([]); // ParentTask对象数组
const activeTaskId = ref(null); // 当前激活（正在进行番茄钟）的任务ID

// 对外暴露的服务接口 (Facade)
const taskService = {
  getActiveTask: () => {
    const allTasks = getAllTasks(); // 展平所有任务
    return allTasks.find(t => t.id === activeTaskId.value);
  },
  setTaskActive: (taskId) => {
    // 首先重置之前激活的任务状态
    const previousActive = taskService.getActiveTask();
    if (previousActive && previousActive.status === TaskStatus.IN_PROGRESS) {
      previousActive.status = TaskStatus.NOT_STARTED;
    }
    // 设置新的激活任务
    const task = findTaskById(taskId);
    if (task) {
      task.status = TaskStatus.IN_PROGRESS;
      activeTaskId.value = taskId;
    }
  },
  updateTaskPomodoroCount: (taskId) => {
    const task = findTaskById(taskId);
    if (task && task instanceof ChildTask) {
      task.completedPomodoros += 1;
      // 当完成一个番茄时，任务状态不变，仍为 IN_PROGRESS 或 NOT_STARTED
      // 只有在用户明确标记完成时，状态才变为 COMPLETED
    }
  },
  // ... 其他必要的查询方法
};
provide('taskService', taskService); // 提供给子组件

2.3 番茄钟模块 (/src/modules/pomodoro.vue)

职责

•   实现 FR-002 所有功能，管理计时器状态机（工作/休息）。

•   接收来自todolist.vue的激活任务，并与之关联。

•   在每个番茄钟（25分钟工作+5分钟休息）自然结束后，自动记录进度。

•   提供计时器的UI控制（开始、暂停、重置/停止）。

计时引擎与状态

// 番茄钟状态枚举
const PomodoroPhase = {
  IDLE: 'idle',
  WORKING: 'work',
  BREAK: 'break'
};

// 响应式状态
const currentPhase = ref(PomodoroPhase.IDLE);
const timeLeft = ref(25 * 60); // 默认25分钟，单位秒
const currentTask = ref(null); // 当前关联的ChildTask对象
const timerId = ref(null);

// 配置
const workDuration = 25 * 60; // 25分钟
const breakDuration = 5 * 60;  // 5分钟

// 注入任务服务
const { taskService } = inject('taskService');

// 核心方法
function startPomodoro(taskId) {
  if (currentPhase.value !== PomodoroPhase.IDLE) return;
  const task = taskService.getTaskById(taskId);
  if (!task) return;
  currentTask.value = task;
  taskService.setTaskActive(taskId);
  switchToPhase(PomodoroPhase.WORKING);
  startTimer();
}
function switchToPhase(phase) {
  currentPhase.value = phase;
  timeLeft.value = phase === PomodoroPhase.WORKING ? workDuration : breakDuration;
}
function onTimerComplete() {
  if (currentPhase.value === PomodoroPhase.WORKING) {
    // 完成一个“工作”阶段，即为完成一个番茄钟
    if (currentTask.value) {
      taskService.updateTaskPomodoroCount(currentTask.value.id);
      // 立即保存进度，满足NF-002.2
      storage.saveTasks(parentTasks.value);
    }
    // 自动进入休息阶段
    switchToPhase(PomodoroPhase.BREAK);
    startTimer();
  } else if (currentPhase.value === PomodoroPhase.BREAK) {
    // 完成一个完整的番茄钟周期（工作+休息）
    switchToPhase(PomodoroPhase.IDLE);
    currentTask.value = null;
    taskService.setTaskActive(null);
  }
}

// 计算今日统计
function getTodayStats() {
  const today = new Date().toDateString();
  const allTasks = getAllTasks(); // 从存储获取所有任务
  const todayPomodoros = allTasks.reduce((sum, task) => {
    // 需要存储模块记录每个番茄钟完成的具体日期
    return sum + (task.completedPomodorosHistory?.filter(h => h.date === today).length || 0);
  }, 0);
  const completedTasksToday = allTasks.filter(t => t.status === TaskStatus.COMPLETED && t.completedDate?.toDateString() === today).length;
  return { todayPomodoros, completedTasksToday };
}

// 计算一周统计
function getWeekStats(startDate) { // startDate 是当周的周日
  const weekData = [];
  for (let i = 0; i < 7; i++) {
    const currentDate = new Date(startDate);
    currentDate.setDate(startDate.getDate() + i);
    const dateStr = currentDate.toDateString();
    const isFuture = currentDate > new Date();
    // 计算该日的番茄钟数
    const dayPomodoros = isFuture ? null : calculatePomodorosForDate(dateStr);
    weekData.push({
      date: currentDate,
      pomodoros: dayPomodoros,
      duration: dayPomodoros === null ? null : dayPomodoros * 30, // 假设每个番茄30分钟
    });
  }
  return weekData;
}

2.5 数据持久化模块 (/src/modules/storage.ts)

职责

  private async _atomicWrite(data: StoredData): Promise<void> {
    const tempPath = getDataPath() + '.tmp';
    const finalPath = getDataPath();
    // 写入临时文件

3. 关键交互流程

1.  启动番茄钟：
    ◦   用户双击一个子任务 -> todolist.vue调用taskService.setTaskActive(taskId) -> 触发事件/状态更新 -> pomodoro.vue监听到活动任务变化，计时器进入“可开始”状态 -> 用户点击底部计时器的“开始”按钮，或设置为自动开始。

2.  完成一个番茄钟：
    ◦   pomodoro.vue计时器WORKING阶段结束 -> 调用taskService.updateTaskPomodoroCount(taskId) -> 触发storage.saveTasks进行自动保存。

    ◦   计时器自动进入BREAK阶段 -> BREAK阶段结束 -> 计时器复位为IDLE，当前活动任务清空。

3.  查看统计：
    ◦   用户点击“统计”按钮 -> 打开stats-modal -> 组件挂载时，stats.vue从storage.ts异步加载数据 -> 计算并渲染图表。
