1. 引言

1.1 编写目的

本说明书定义todoList系统的轻量化架构设计，采用模块化Vue3组件结构，旨在最小化复杂度、最大化开发效率。本文档依据"一人公司开发策略"重新设计，所有业务逻辑与UI组件内聚于同一模块文件中，实现文件内部传递，避免不必要的消息传递机制。

1.2 系统概述

todoList是一款专注于个人学习的轻量化小型桌面应用程序。采用"功能内聚、文件简化"的设计理念，将传统的分层架构压缩为扁平化模块结构，每个功能模块独立封装完整的业务逻辑和UI组件，通过最小化的外观接口进行必要通信。

1.3 设计约束

• 架构简化：放弃传统分层架构，采用模块化组件设计

• 最小化依赖：模块间仅通过定义清晰的接口进行必要通信

• 文件最小化：每个功能模块尽量在单个文件中实现完整的逻辑和UI

• 零IPC开销：所有组件间通信在同一Vue应用实例内完成

• 开发效率优先：为单人开发优化，避免过度工程化

2. 系统架构设计

2.1 物理架构视图

系统采用单节点桌面应用架构。应用通过Tauri框架打包，分发为独立可执行文件。所有Vue3组件、TypeScript模块和业务逻辑运行在同一个Webview进程中，共享内存空间，无进程间通信开销。

技术栈构成：
• 前端框架：Vue 3 + Composition API + TypeScript

• 构建工具：Vite

• 桌面打包：Tauri

• 数据持久化：本地JSON文件存储

• UI风格：原生CSS + Tailwind CSS可选

• 数据可视化：Chart.js Lite 或 Native Chart

2.2 逻辑架构视图（模块化扁平架构）

为实现高内聚、低复杂度，系统采用模块化扁平架构：

应用主入口 (App.vue)
    ├── 任务管理模块 (TodoList.vue)     # 包含完整任务管理逻辑和UI
    ├── 番茄钟模块 (Pomodoro.vue)       # 包含完整计时逻辑和UI
    └── 统计分析模块 (Statistics.vue)   # 包含完整统计逻辑和UI
        └── 数据存储模块 (storage.ts)   # 纯TS模块，无UI


设计原则：
1. 功能完整内聚：每个.vue文件既是组件又是模块，包含该功能的完整业务逻辑
2. 最小化接口：模块间仅通过少量明确定义的方法进行通信
3. 无中间层：直接调用，避免复杂的消息传递机制
4. 状态自管理：每个模块管理自己的响应式状态
5. 外观模式应用：复杂模块对外暴露简化的服务接口

<script setup lang="ts">
import { ref, reactive, computed, provide } from 'vue';
import { Task, TaskStatus, TaskPriority } from '@/types';
import { storage } from './storage';

// 响应式状态
const tasks = ref<Task[]>([]);
const categories = ref<Category[]>([]);
const activeTaskId = ref<string | null>(null);
const selectedCategory = ref<string | null>(null);

// 对外服务接口（外观模式）
const taskService = {
  // 获取当前活动任务
  getActiveTask: (): Task | null => {
    return tasks.value.find(task => task.id === activeTaskId.value) || null;
  },
  
  // 设置活动任务
  setTaskActive: (taskId: string) => {
    activeTaskId.value = taskId;
  },
  
  // 更新任务番茄钟数量
  updateTaskPomodoroCount: (taskId: string, countDelta: number) => {
    const task = tasks.value.find(t => t.id === taskId);
    if (task) {
      task.usedPomos = Math.max(0, task.usedPomos + countDelta);
      if (task.usedPomos > 0 && task.status === 'not_started') {
        task.status = 'in_progress';
      }
      // 自动保存
      storage.saveTasks(tasks.value);
    }
  },
  
  // 获取所有任务
  getAllTasks: () => [...tasks.value],
  
  // 获取按分类筛选的任务
  getTasksByCategory: (categoryId: string) => 
    tasks.value.filter(task => task.parentId === categoryId),
  
  // 创建新任务
  createTask: (taskData: Partial<Task>): Task => {
    const newTask: Task = {
      id: generateUUID(),
      parentId: taskData.parentId || null,
      title: taskData.title || '新任务',
      description: taskData.description,
      dueDate: taskData.dueDate,
      createdAt: Date.now(),
      usedPomos: 0,
      status: 'not_started',
      priority: taskData.priority || 'medium'
    };
    
    tasks.value.push(newTask);
    storage.saveTasks(tasks.value);
    return newTask;
  },
  
  // 删除任务
  deleteTask: (taskId: string) => {
    tasks.value = tasks.value.filter(task => task.id !== taskId);
    storage.saveTasks(tasks.value);
  },
  
  // 更新任务状态
  updateTaskStatus: (taskId: string, status: TaskStatus) => {
    const task = tasks.value.find(t => t.id === taskId);
    if (task) {
      task.status = status;
      storage.saveTasks(tasks.value);
    }
  }
};

// 将服务提供给子组件
provide('taskService', taskService);

// 初始化加载数据
const loadTasks = async () => {
  const data = await storage.loadData();
  tasks.value = data.tasks;
};
</script>

3.2.2 番茄钟引擎实现

<script setup lang="ts">
import { ref, reactive, inject, onUnmounted } from 'vue';
import { PomodoroRecord } from '@/types';
import { storage } from './storage';

// 依赖注入获取任务服务
const taskService = inject('taskService');

// 番茄钟状态
const pomodoroState = reactive({
  status: 'idle' as 'idle' | 'work' | 'break' | 'paused',
  currentTaskId: null as string | null,
  startTime: 0,
  remainingTime: 0,
  timerId: null as number | null
});

// 配置
const settings = ref({
  workDuration: 25 * 60, // 25分钟，单位秒
  breakDuration: 5 * 60   // 5分钟，单位秒
});

// 番茄钟引擎
class PomodoroEngine {
  start(taskId?: string) {
    if (pomodoroState.status !== 'idle') return;
    
    pomodoroState.currentTaskId = taskId || null;
    pomodoroState.status = 'work';
    pomodoroState.startTime = Date.now();
    pomodoroState.remainingTime = settings.value.workDuration;
    
    pomodoroState.timerId = setInterval(() => this.tick(), 1000);
    
    // 通知任务模块更新状态
    if (taskId && taskService) {
      taskService.setTaskActive(taskId);
    }
  }
  
  pause() {
    if (pomodoroState.status === 'work' || pomodoroState.status === 'break') {
      pomodoroState.status = 'paused';
      if (pomodoroState.timerId) {
        clearInterval(pomodoroState.timerId);
        pomodoroState.timerId = null;
      }
    }
  }
  
  resume() {
    if (pomodoroState.status === 'paused') {
      pomodoroState.status = pomodoroState.remainingTime > 0 ? 'work' : 'break';
      pomodoroState.timerId = setInterval(() => this.tick(), 1000);
    }
  }
  
  stop() {
    if (pomodoroState.timerId) {
      clearInterval(pomodoroState.timerId);
      pomodoroState.timerId = null;
    }
    
    // 重置状态
    pomodoroState.status = 'idle';
    pomodoroState.currentTaskId = null;
    pomodoroState.remainingTime = 0;
  }
  
  private tick() {
    pomodoroState.remainingTime--;
    
    if (pomodoroState.remainingTime <= 0) {
      this.onTimerComplete();
    }
  }
  
  private async onTimerComplete() {
    if (pomodoroState.timerId) {
      clearInterval(pomodoroState.timerId);
      pomodoroState.timerId = null;
    }
    
    if (pomodoroState.status === 'work') {
      // 工作完成，记录番茄钟
      await this.recordPomodoroCompletion();
      
      // 开始休息
      pomodoroState.status = 'break';
      pomodoroState.remainingTime = settings.value.breakDuration;
      pomodoroState.startTime = Date.now();
      pomodoroState.timerId = setInterval(() => this.tick(), 1000);
      
    } else if (pomodoroState.status === 'break') {
      // 休息完成，一个完整番茄钟结束
      pomodoroState.status = 'idle';
    }
  }
  
  private async recordPomodoroCompletion() {
    if (pomodoroState.currentTaskId) {
      // 更新任务番茄钟数量
      if (taskService) {
        taskService.updateTaskPomodoroCount(pomodoroState.currentTaskId, 1);
      }
      
      // 保存番茄钟记录
      const record: PomodoroRecord = {
        id: generateUUID(),
        taskId: pomodoroState.currentTaskId,
        startTime: pomodoroState.startTime,
        endTime: Date.now(),
        duration: settings.value.workDuration,
        type: 'work',
        completed: true
      };
      
      await storage.savePomodoroRecord(record);
    }
  }
}

const engine = new PomodoroEngine();

// 组件卸载时清理计时器
onUnmounted(() => {
  if (pomodoroState.timerId) {
    clearInterval(pomodoroState.timerId);
  }
});
</script>

              record.completed
  );
  
  const todayTasks = tasks.filter(task => 
    task.status === 'completed' && 
    task.completedAt && 
    task.completedAt >= todayStart && 
    task.completedAt < tomorrowStart
  );
  
  return {
    completedPomos: todayRecords.filter(r => r.type === 'work').length,
    completedTasks: todayTasks.length,
    totalFocusTime: todayRecords.reduce((sum, r) => sum + r.duration, 0),
    averageFocusTime: todayRecords.length > 0 
      ? todayRecords.reduce((sum, r) => sum + r.duration, 0) / todayRecords.length 
      : 0
  };
};

// 计算本周统计
const calculateWeekStats = (records: PomodoroRecord[]) => {
  const now = new Date();
  const startOfWeek = getStartOfWeek(now); // 周日为一周开始
  const endOfWeek = new Date(startOfWeek);
  endOfWeek.setDate(startOfWeek.getDate() + 7);
  
  const weekRecords = records.filter(
    record => record.startTime >= startOfWeek.getTime() && 
              record.startTime < endOfWeek.getTime() &&
              record.completed
  );
  
  // 按天分组
  const dailyStats: DailyStat[] = [];
  for (let i = 0; i < 7; i++) {
    const day = new Date(startOfWeek);
    day.setDate(startOfWeek.getDate() + i);
    
    const dayRecords = weekRecords.filter(record => {
      const recordDate = new Date(record.startTime);
      return recordDate.getDate() === day.getDate() &&
             recordDate.getMonth() === day.getMonth() &&
             recordDate.getFullYear() === day.getFullYear();
    });
    
    dailyStats.push({
      date: day,
      pomoCount: dayRecords.filter(r => r.type === 'work').length,
      focusTime: dayRecords.reduce((sum, r) => sum + r.duration, 0)
    });
  }
  
  return {
    dailyStats,
    totalPomos: weekRecords.filter(r => r.type === 'work').length,
    totalFocusTime: weekRecords.reduce((sum, r) => sum + r.duration, 0)
  };
};

/**
 * 数据存储服务
 */
export const storage = {
  private dataPath: string = '';
  private isSaving: boolean = false;
  private saveQueue: (() => Promise<void>)[] = [];
  
  // 初始化
  async initialize(): Promise<void> {
    // 获取应用数据目录
    const appDataDir = await getAppDataDir();
    this.dataPath = path.join(appDataDir, 'todoList', 'data.json');
    
    // 确保目录存在
    await ensureDir(path.dirname(this.dataPath));
    
    // 检查并修复数据文件
    await this.checkAndRepairData();
  },
  
  // 加载所有数据
  async loadData(): Promise<AppData> {
    try {
      if (!await this.fileExists(this.dataPath)) {
        return this.getDefaultData();
      }
      
      const data = await this.readFile(this.dataPath);
      const parsed = JSON.parse(data) as AppData;
      
      // 数据验证和修复
      return this.validateAndRepairData(parsed);
      
    } catch (error) {
      console.error('Failed to load data:', error);
      return this.getDefaultData();
    }
  },
  
  // 保存任务数据（自动保存）
  async saveTasks(tasks: Task[]): Promise<void> {
    return this.enqueueSave(async () => {
      const data = await this.loadData();
      data.tasks = tasks;
      data.lastModified = Date.now();
      await this.saveData(data);
    });
  },
  
  // 保存番茄钟记录
  async savePomodoroRecord(record: PomodoroRecord): Promise<void> {
    return this.enqueueSave(async () => {
      const data = await this.loadData();
      data.records.push(record);
      data.lastModified = Date.now();
      await this.saveData(data);
    });
  },
  
  // 保存应用设置
  async saveSettings(settings: AppSettings): Promise<void> {
    return this.enqueueSave(async () => {
      const data = await this.loadData();
      data.settings = settings;
      data.lastModified = Date.now();
      await this.saveData(data);
    });
  },
  
  // 原子化保存
  private async saveData(data: AppData): Promise<void> {
    const tempPath = this.dataPath + '.tmp';
    const jsonData = JSON.stringify(data, null, 2);
    
    try {
      // 1. 写入临时文件
      await this.writeFile(tempPath, jsonData);
      
      // 2. 验证写入
      const tempContent = await this.readFile(tempPath);
      if (tempContent !== jsonData) {
        throw new Error('File write verification failed');
      }
      
      // 3. 原子性替换
      await this.renameFile(tempPath, this.dataPath);
      
      // 4. 可选备份
      if (data.settings.backupEnabled) {
        await this.createBackup(data);
      }
      
    } catch (error) {
      console.error('Failed to save data:', error);
      throw error;
    }
  },
  
  // 数据验证和修复
  private validateAndRepairData(data: AppData): AppData {
    const repaired = { ...data };
    
    // 版本兼容性
    if (!repaired.version) {
      repaired.version = '1.0.0';
    }
    
    // 任务数据修复
    if (!Array.isArray(repaired.tasks)) {
      repaired.tasks = [];
    } else {
      repaired.tasks = repaired.tasks.map(task => ({
        ...task,
        id: task.id || generateUUID(),
        createdAt: task.createdAt || Date.now(),
        usedPomos: task.usedPomos || 0,
        status: task.status || 'not_started',
        priority: task.priority || 'medium'
      }));
    }
    
    // 修复状态不一致的任务
    repaired.tasks.forEach(task => {
      if (task.status === 'in_progress' && task.usedPomos === 0) {
        task.status = 'not_started';
      }
    });
    
    // 记录数据修复
    if (!Array.isArray(repaired.records)) {
      repaired.records = [];
    }
    
    // 设置默认值
    if (!repaired.settings) {
      repaired.settings = this.getDefaultSettings();
    }
    
    return repaired;
  },
  
  // 检查并修复数据
  private async checkAndRepairData(): Promise<void> {
    try {
      if (await this.fileExists(this.dataPath)) {
        const data = await this.loadData();
        await this.saveData(data); // 保存修复后的数据
      }
    } catch (error) {
      console.warn('Data repair completed with warnings:', error);
    }
  },
  
  // 排队保存（防止并发写入）
  private async enqueueSave(saveFn: () => Promise<void>): Promise<void> {
    return new Promise((resolve, reject) => {
      this.saveQueue.push(async () => {
        try {
          await saveFn();
          resolve();
        } catch (error) {
          reject(error);
        }
      });
      
      if (!this.isSaving) {
        this.processSaveQueue();
      }
    });
  },
  
  private async processSaveQueue(): Promise<void> {
    if (this.saveQueue.length === 0) {
      this.isSaving = false;
      return;
    }
    
    this.isSaving = true;
    const saveFn = this.saveQueue.shift();
    
    if (saveFn) {
      try {
        await saveFn();
      } finally {
        // 延迟处理下一个，防止过频保存
        setTimeout(() => this.processSaveQueue(), 100);
      }
    }
  }
};

// 数据加密（可选功能，根据NF-004.1调整）
export const dataEncryption = {
  private encryptionKey: string | null = null,
  
  // 启用加密
  enableEncryption(password: string): void {
    // 使用Web Crypto API进行加密
    this.encryptionKey = this.deriveKey(password);
  },
  
  // 加密数据
  async encrypt(data: string): Promise<string> {
    if (!this.encryptionKey) return data;
    
    try {
      const encoder = new TextEncoder();
      const dataBuffer = encoder.encode(data);
      
      // 生成随机IV
      const iv = crypto.getRandomValues(new Uint8Array(12));
      
      // 加密
      const encryptedBuffer = await crypto.subtle.encrypt(
        {
          name: 'AES-GCM',
          iv: iv
        },
        this.encryptionKey!,
        dataBuffer
      );
      
      // 合并IV和加密数据
      const encryptedArray = new Uint8Array(encryptedBuffer);
      const result = new Uint8Array(iv.length + encryptedArray.length);
      result.set(iv, 0);
      result.set(encryptedArray, iv.length);
      
      return btoa(String.fromCharCode(...result));
      
    } catch (error) {
      console.error('Encryption failed:', error);
      return data;
    }
  },
  
  // 解密数据
  async decrypt(encryptedData: string): Promise<string> {
    if (!this.encryptionKey) return encryptedData;
    
    try {
      // 解码base64
      const binaryString = atob(encryptedData);
      const bytes = new Uint8Array(binaryString.length);
      for (let i = 0; i < binaryString.length; i++) {
        bytes[i] = binaryString.charCodeAt(i);
      }
      
      // 提取IV和加密数据
      const iv = bytes.slice(0, 12);
      const encrypted = bytes.slice(12);
      
      // 解密
      const decryptedBuffer = await crypto.subtle.decrypt(
        {
          name: 'AES-GCM',
          iv: iv
        },
        this.encryptionKey!,
        encrypted
      );
      
      const decoder = new TextDecoder();
      return decoder.decode(decryptedBuffer);
      
    } catch (error) {
      console.error('Decryption failed:', error);
      return encryptedData;
    }
  },
  
  // 派生密钥
  private deriveKey(password: string): CryptoKey {
    // 使用PBKDF2派生密钥
    const encoder = new TextEncoder();
    const salt = encoder.encode('todoList-salt'); // 应使用固定或存储的salt
    
    return crypto.subtle.importKey(
      'raw',
      encoder.encode(password),
      { name: 'PBKDF2' },
      false,
      ['deriveKey']
    ).then(baseKey => 
      crypto.subtle.deriveKey(
        {
          name: 'PBKDF2',
          salt: salt,
          iterations: 100000,
          hash: 'SHA-256'
        },
        baseKey,
        { name: 'AES-GCM', length: 256 },
        false,
        ['encrypt', 'decrypt']
      )
    );
  }
};

export const logger = {
  private enabled: true,
  private level: ErrorLevel.INFO,
  private logFile: string = '',
  
  async initialize(): Promise<void> {
    const logDir = await getLogDir();
    this.logFile = path.join(logDir, 'todoList.log');
    
    // 限制日志文件大小
    await this.rotateLogIfNeeded();
  },
  
  log(level: ErrorLevel, message: string, data?: any): void {
    if (!this.enabled || !this.shouldLog(level)) return;
    
    const timestamp = new Date().toISOString();
    const logEntry = {
      timestamp,
      level,
      message,
      data: data ? JSON.stringify(data) : undefined
    };
    
    // 控制台输出
    const consoleMethod = this.getConsoleMethod(level);
    console[consoleMethod](`[${level}] ${message}`, data || '');
    
    // 文件记录
    this.writeToFile(JSON.stringify(logEntry) + '\n');
  },
  
  private async writeToFile(content: string): Promise<void> {
    try {
      await appendFile(this.logFile, content);
    } catch (error) {
      console.error('Failed to write to log file:', error);
    }
  },
  
  private shouldLog(level: ErrorLevel): boolean {
    const levels = Object.values(ErrorLevel);
    const currentLevelIndex = levels.indexOf(this.level);
    const messageLevelIndex = levels.indexOf(level);
    return messageLevelIndex >= currentLevelIndex;
  },
  
  private getConsoleMethod(level: ErrorLevel): keyof Console {
    switch (level) {
      case ErrorLevel.DEBUG: return 'debug';
      case ErrorLevel.INFO: return 'info';
      case ErrorLevel.WARN: return 'warn';
      case ErrorLevel.ERROR: return 'error';
      case ErrorLevel.FATAL: return 'error';
      default: return 'log';
    }
  }
};
